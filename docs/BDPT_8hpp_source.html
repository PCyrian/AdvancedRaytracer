124 inline Math::Vector3D refract_vec(const Math::Vector3D& incident_I_pointing_towards_surface, const Math::Vector3D& N_interface_from_inc_to_trans, double eta_ratio_incident_over_transmitted) {
125 double cos_theta_i = -incident_I_pointing_towards_surface.dot(N_interface_from_inc_to_trans);
126 cos_theta_i = std::max(-1.0, std::min(1.0, cos_theta_i)); // Clamp
127
128 double sin2_theta_t = eta_ratio_incident_over_transmitted * eta_ratio_incident_over_transmitted * (1.0 - cos_theta_i * cos_theta_i);
129
130 if (sin2_theta_t > 1.000001) return Math::Vector3D(0,0,0);
131
132 double cos_theta_t = std::sqrt(std::max(0.0, 1.0 - sin2_theta_t));
133 return incident_I_pointing_towards_surface * eta_ratio_incident_over_transmitted +
134 N_interface_from_inc_to_trans * (eta_ratio_incident_over_transmitted * cos_theta_i - cos_theta_t);
135 }

144 inline double fresnel_schlick_scalar(double cos_incident_angle_wrt_normal, double n1_incident_medium, double n2_transmitted_medium) {
145 if (n1_incident_medium == n2_transmitted_medium) return 0.0;
146
147 double R0_num = (n1_incident_medium - n2_transmitted_medium);
148 double R0_den = (n1_incident_medium + n2_transmitted_medium);
149 if (std::abs(R0_den) < PDF_EPSILON) return 1.0;
150 double R0 = R0_num / R0_den;
151 R0 = R0 * R0;
152
153 // Explicitly check for Total Internal Reflection (TIR) if ray is coming from a denser medium.
154 if (n1_incident_medium > n2_transmitted_medium) {
155 double sin2_theta_i = 1.0 - cos_incident_angle_wrt_normal * cos_incident_angle_wrt_normal;
156 if (sin2_theta_i < 0.0) sin2_theta_i = 0.0; // Clamp
157 double n_ratio_sq = (n1_incident_medium / n2_transmitted_medium) * (n1_incident_medium / n2_transmitted_medium);
158 double sin2_theta_t_check = n_ratio_sq * sin2_theta_i;
159 if (sin2_theta_t_check > 1.0) return 1.0; // Total Internal Reflection.
160 }
161
162 double term = 1.0 - std::max(0.0, cos_incident_angle_wrt_normal);
163 return R0 + (1.0 - R0) * term * term * term * term * term; // Schlick's approximation.
164 }

172 inline Math::Vector3D sample_cosine_hemisphere_bdpt(const Math::Vector3D& normal, double& pdf_out) {
173 double r1 = local_rand01();
174 double r2 = local_rand01();
175
176 double sin_theta_sample = std::sqrt(r1);
177 double cos_theta_sample = std::sqrt(1.0 - r1);
178 double phi = 2.0 * M_PI * r2;
179
180 Math::Vector3D w = normal;
181 Math::Vector3D t_candidate = (std::abs(w.x) > 0.1 ? Math::Vector3D(0, 1, 0) : Math::Vector3D(1, 0, 0));
182 Math::Vector3D u_basis = safe_normalize(t_candidate.cross(w));
183 Math::Vector3D v_basis = w.cross(u_basis);
184
185 pdf_out = cos_theta_sample * BDPT::ONE_OVER_PI;
186 if (pdf_out < PDF_EPSILON) pdf_out = PDF_EPSILON;
187
188 return safe_normalize(u_basis * std::cos(phi) * sin_theta_sample +
189 v_basis * std::sin(phi) * sin_theta_sample +
190 w * cos_theta_sample);
191 }

198 inline Math::Vector3D sample_uniform_sphere_direction_bdpt(double& pdf_out) {
199 double z = 1.0 - 2.0 * local_rand01();
200 double r_xy = std::sqrt(std::max(0.0, 1.0 - z*z));
201 double phi = 2.0 * M_PI * local_rand01();
202 pdf_out = BDPT::ONE_OVER_FOUR_PI;
203 return Math::Vector3D(r_xy * std::cos(phi), r_xy * std::sin(phi), z);
204 }

240 Math::Vector3D eval_bsdf(const RayTracer::HitRecord& rec, const Math::Vector3D& wo_world, const Math::Vector3D& wi_world,
241 [[maybe_unused]] const Math::Vector3D& N_geom, [[maybe_unused]] bool wi_is_transmission_hint) {
242 const RayTracer::Material& M = rec.material;
243 Math::Vector3D albedo = get_material_albedo_from_rec(M, rec);
244
245 Math::Vector3D N_eff = rec.normal;
246 if (N_eff.dot(wo_world) < 0.0) N_eff = N_eff * -1.0;
247
248 double N_eff_dot_wi = N_eff.dot(wi_world);
249
250 // --- Opaque Non-Metallic (Diffuse/Glossy Dielectric) ---
251 if (M.metallic < 0.5 && M.transparency < 0.5) {
252 if (N_eff_dot_wi > 0.0) {
253 return albedo * BDPT::ONE_OVER_PI;
254 }
255 }
256 // --- Metallic ---
257 else if (M.metallic >= 0.5) {
258 if (M.roughness <= 0.05f) return Math::Vector3D(0,0,0); // Perfect mirrors are delta events, eval is 0 for arbitrary wi.
259 if (N_eff_dot_wi > 0.0) {
260 return M.specular * BDPT::ONE_OVER_PI;
261 }
262 }
263 // --- Transparent Dielectric ---
264 else if (M.transparency >= 0.5) {
265 if (M.roughness <= 0.05f) return Math::Vector3D(0,0,0); // Perfect dielectrics
266
267 // Approximate BSDF for rough dielectrics
268 bool wo_is_exiting_material = rec.normal.dot(wo_world) > 0.0;
269 double n1 = wo_is_exiting_material ? M.ior : 1.0;
270 double n2 = wo_is_exiting_material ? 1.0 : M.ior;
271
272 double cos_theta_o_abs = std::abs(N_eff.dot(wo_world)); // Cosine of outgoing angle w.r.t effective normal.
273 cos_theta_o_abs = std::min(1.0, std::max(0.0, cos_theta_o_abs)); // Clamp
274
275 double R_fresnel = fresnel_schlick_scalar(cos_theta_o_abs, n1, n2);
276
277 if (N_eff_dot_wi * N_eff.dot(wo_world) > 0) {
278 return M.specular * R_fresnel * BDPT::ONE_OVER_PI;
279 } else { // Transmission.
280 Math::Vector3D transmission_tint = albedo;
281 return transmission_tint * (1.0 - R_fresnel) * BDPT::ONE_OVER_PI;
282 }
283 }
284 return Math::Vector3D(0,0,0);
285 }

299 void sample_bsdf(const RayTracer::HitRecord& rec, const Math::Vector3D& wo_world,
300 Math::Vector3D& wi_sampled_world, double& pdf_solid_angle_wi,
301 Math::Vector3D& bsdf_term_value, bool& scattered_is_delta) {
302
303 const RayTracer::Material& M = rec.material;
304 Math::Vector3D albedo_val = get_material_albedo_from_rec(M, rec);
305
306 scattered_is_delta = false;
307 bsdf_term_value = Math::Vector3D(0,0,0);
308 pdf_solid_angle_wi = 0.0;
309 wi_sampled_world = Math::Vector3D(0,0,0);
310
311 Math::Vector3D V_incident_path = wo_world * -1.0;
312
313 bool is_entering_object_boundary = V_incident_path.dot(rec.normal) < 0.0;
314
315 Math::Vector3D N_shading_oriented;
316 Math::Vector3D N_interface_for_snell;
317 double n_incident_medium_ior;
318 double n_transmitted_medium_ior;
319
320 double base_dielectric_ior_for_highlights = 1.5;
321
322 if (is_entering_object_boundary) {
323 N_shading_oriented = rec.normal;
324 N_interface_for_snell = rec.normal;
325 n_incident_medium_ior = 1.0;
326 n_transmitted_medium_ior = M.ior;
327 } else {
328 N_shading_oriented = rec.normal * -1.0;
329 N_interface_for_snell = rec.normal * -1.0;
330 n_incident_medium_ior = M.ior;
331 n_transmitted_medium_ior = 1.0;
332 }
333
334 double N_dot_V = std::min(1.0, std::max(0.0, -V_incident_path.dot(N_shading_oriented)));
335 Math::Vector3D material_specular_f0_tint = M.specular;
336
337
338 if (M.metallic >= 0.5f) { // --- METALLIC MATERIAL ---
339 scattered_is_delta = (M.roughness < 0.05f); // Smooth metals are perfect mirrors (delta).
340 wi_sampled_world = reflect_vec(V_incident_path, N_shading_oriented); // Metals only reflect.
341 bsdf_term_value = material_specular_f0_tint;
342
343 if (scattered_is_delta) {
344 pdf_solid_angle_wi = 1.0;
345 } else {
346 sample_cosine_hemisphere_bdpt(N_shading_oriented, pdf_solid_angle_wi);
347 }
348
349 } else if (M.transparency >= 0.5f) { // --- TRANSPARENT DIELECTRIC (e.g., Glass, Water) ---
350 scattered_is_delta = (M.roughness < 0.05f);
351
352 double fresnel_reflect_prob = fresnel_schlick_scalar(N_dot_V, n_incident_medium_ior, n_transmitted_medium_ior);
353
354 if (local_rand01() < fresnel_reflect_prob) {
355 wi_sampled_world = reflect_vec(V_incident_path, N_shading_oriented);
356 bsdf_term_value = material_specular_f0_tint;
357 } else {
358 double eta_ratio = n_incident_medium_ior / n_transmitted_medium_ior;
359 wi_sampled_world = refract_vec(V_incident_path, N_interface_for_snell, eta_ratio);
360
361 if (wi_sampled_world.dot(wi_sampled_world) < GEOMETRY_EPSILON * GEOMETRY_EPSILON) {
362 wi_sampled_world = reflect_vec(V_incident_path, N_shading_oriented);
363 bsdf_term_value = material_specular_f0_tint;
364 } else { // Successful transmission.
365 bsdf_term_value = albedo_val;
366 }
367 }
368 // PDF for delta reflection/transmission.
369 if (scattered_is_delta) {
370 pdf_solid_angle_wi = 1.0;
371 } else {
372 // For rough dielectrics, ideally use microfacet model for sampling.
373 sample_cosine_hemisphere_bdpt(N_shading_oriented, pdf_solid_angle_wi);
374 }
375
376 } else { // --- OPAQUE NON-METALLIC (e.g., Diffuse, Plastic, Matte) ---
377 double physical_fresnel_prob_for_highlights = fresnel_schlick_scalar(N_dot_V, 1.0, base_dielectric_ior_for_highlights);
378
379 if (material_specular_f0_tint.dot(material_specular_f0_tint) < 0.001f && M.roughness > 0.5f) {
380 physical_fresnel_prob_for_highlights = 0.0; // Force diffuse path.
381 }
382
383 if (local_rand01() < physical_fresnel_prob_for_highlights) { // Sample Specular Lobe (for highlights).
384 scattered_is_delta = (M.roughness < 0.05f); // Glossy highlights are non-delta if roughness > ~0.05.
385 wi_sampled_world = reflect_vec(V_incident_path, N_shading_oriented);
386 bsdf_term_value = material_specular_f0_tint;
387
388 if (scattered_is_delta) {
389 pdf_solid_angle_wi = 1.0;
390 } else {
391 sample_cosine_hemisphere_bdpt(N_shading_oriented, pdf_solid_angle_wi);
392 }
393 } else { // Sample Diffuse Lobe.
394 scattered_is_delta = false;
395 wi_sampled_world = sample_cosine_hemisphere_bdpt(N_shading_oriented, pdf_solid_angle_wi);
396 if (pdf_solid_angle_wi > PDF_EPSILON) {
397 bsdf_term_value = albedo_val;
398 } else {
399 bsdf_term_value = Math::Vector3D(0,0,0);
400 wi_sampled_world = Math::Vector3D(0,0,0);
401 }
402 }
403 }
404
405 if (wi_sampled_world.dot(wi_sampled_world) < GEOMETRY_EPSILON * GEOMETRY_EPSILON ) {
406 pdf_solid_angle_wi = 0.0;
407 bsdf_term_value = Math::Vector3D(0,0,0);
408 scattered_is_delta = true;
409 } else if (pdf_solid_angle_wi < PDF_EPSILON && !scattered_is_delta) {
410 bsdf_term_value = Math::Vector3D(0,0,0);
411 }
412 if (!scattered_is_delta && pdf_solid_angle_wi < PDF_EPSILON && bsdf_term_value.dot(bsdf_term_value) > GEOMETRY_EPSILON) {
413 pdf_solid_angle_wi = PDF_EPSILON;
414 }
415 }

429 int generate_subpath_fixed(const RayTracer::Scene& scene, RayTracer::Ray current_ray, int max_bounces,
430 Path& path_vertices, bool is_camera_path, [[maybe_unused]] const Utils::RenderParams& params) {
431 path_vertices.clear();
432 Math::Vector3D beta(1.0, 1.0, 1.0);
433 double pdf_fwd_prev_bounce_solid_angle = 1.0;
434
435 for (int bounce = 0; bounce <= max_bounces; ++bounce) {
436 RayTracer::HitRecord rec_hit;
437 double t_hit = std::numeric_limits<double>::max();
438 bool hit_surface = scene.hit(current_ray, t_hit, rec_hit);
439
440 RayTracer::Vertex v;
441 v.type = RayTracer::VertexType::Surface;
442 v.wi = current_ray.direction * -1.0;
443 v.beta = beta;
444 v.pdfFwd = pdf_fwd_prev_bounce_solid_angle;
445 v.is_point_light_source = false;
446
447 if (!hit_surface) {
448 if (is_camera_path) {
449 RayTracer::Vertex sky_v;
450 sky_v.type = RayTracer::VertexType::Surface;
451 sky_v.rec.t = std::numeric_limits<double>::infinity();
452 sky_v.rec.point = current_ray.origin + current_ray.direction * 1e10;
453 sky_v.rec.normal = current_ray.direction * -1.0;
454 sky_v.rec.material.emission = Skybox::skyGradient(current_ray.direction);
455 sky_v.rec.material.emissionStrength = 1.0f;
456 sky_v.rec.material.albedo = Math::Vector3D(0,0,0);
457 sky_v.wi = current_ray.direction * -1.0f;
458 sky_v.beta = beta;
459 sky_v.pdfFwd = pdf_fwd_prev_bounce_solid_angle;
460 sky_v.delta = false;
461 sky_v.is_point_light_source = false;
462 path_vertices.push_back(sky_v);
463 }
464 break; // End of path if ray misses
465 }
466
467 v.rec = rec_hit;
468
469 // Determine if the BSDF at this vertex is a delta function (perfect specular).
470 bool bsdf_at_this_vertex_is_delta = false;
471 const auto& M_curr = rec_hit.material;
472 if (M_curr.metallic >= 0.5f && M_curr.roughness < 0.05f) { // Smooth metal.
473 bsdf_at_this_vertex_is_delta = true;
474 } else if (M_curr.transparency >= 0.5f && M_curr.roughness < 0.05f) { // Smooth dielectric.
475 bsdf_at_this_vertex_is_delta = true;
476 }
477 v.delta = bsdf_at_this_vertex_is_delta;
478
479 path_vertices.push_back(v);
480
481 if (v.rec.material.emission.dot(v.rec.material.emission) * v.rec.material.emissionStrength > GEOMETRY_EPSILON) {
482 if (!is_camera_path && bounce == 0) {
483 } else {
484 break;
485 }
486 }
487
488 // Sample next direction from BSDF at the current vertex.
489 Math::Vector3D wi_sampled_next_direction; // Sampled direction for the next ray.
490 double pdf_for_wi_sampled_next_direction; // PDF of sampling that direction.
491 Math::Vector3D bsdf_throughput_factor; // BSDF value (f_s * albedo/specular) for the sampled direction.
492 bool next_ray_will_be_from_delta_scatter; // Was the BSDF sample a delta event?
493
494 sample_bsdf(rec_hit, current_ray.direction * -1.0,
495 wi_sampled_next_direction, pdf_for_wi_sampled_next_direction,
496 bsdf_throughput_factor, next_ray_will_be_from_delta_scatter);
497
498 if ( (pdf_for_wi_sampled_next_direction < PDF_EPSILON && !next_ray_will_be_from_delta_scatter) ||
499 bsdf_throughput_factor.dot(bsdf_throughput_factor) < THROUGHPUT_EPSILON_SQ ) {
500 break;
501 }
502
503 beta = beta * bsdf_throughput_factor;
504
505 if (!path_vertices.empty()) {
506 path_vertices.back().pdfRev = pdf_for_wi_sampled_next_direction;
507 }
508
509 // Update PDF for the next bounce.
510 pdf_fwd_prev_bounce_solid_angle = pdf_for_wi_sampled_next_direction;
511
512 // Create next ray.
513 current_ray = RayTracer::Ray(rec_hit.point + wi_sampled_next_direction * RAY_OFFSET_EPSILON,
514 wi_sampled_next_direction);
515
516 // Russian Roulette for path termination.
517 if (bounce >= BDPT::RR_START_DEPTH) {
518 double max_beta_comp = std::max({beta.x, beta.y, beta.z});
519 if (max_beta_comp < 0.0) max_beta_comp = 0.0;
520 double p_continue = std::min(0.95, max_beta_comp);
521
522 if (local_rand01() > p_continue) { break; }
523 if (p_continue < PDF_EPSILON) break;
524 beta = beta / p_continue;
525 }
526
527 if (beta.dot(beta) < THROUGHPUT_EPSILON_SQ) break;
528 }
529 return path_vertices.size();
530 }

544 bool sample_light_source_fixed(const RayTracer::Scene& scene, RayTracer::Ray& light_ray_out, Math::Vector3D& Le_value_out,
545 double& pdf_pos_area_light_out, double& pdf_emit_dir_solid_angle_out,
546 RayTracer::Vertex& light_vertex_out) {
547 light_vertex_out.is_point_light_source = false;
548
549 std::vector<const RayTracer::Primitive*> emissive_prims;
550 for(const auto& obj_sptr : scene.objects) {
551 const auto& mat = obj_sptr->getMaterial();
552 if ((mat.emission.dot(mat.emission) > GEOMETRY_EPSILON) && mat.emissionStrength > GEOMETRY_EPSILON) {
553 emissive_prims.push_back(obj_sptr.get());
554 }
555 }
556
557 bool has_explicit_lights = !scene.lights.empty();
558 bool has_emissive_prims = !emissive_prims.empty();
559
560 if (!has_explicit_lights && !has_emissive_prims) return false;
561
562 double prob_pick_explicit_category = 0.0;
563 if(has_explicit_lights && has_emissive_prims) prob_pick_explicit_category = 0.5;
564 else if(has_explicit_lights) prob_pick_explicit_category = 1.0;
565
566 bool sample_from_explicit_list = local_rand01() < prob_pick_explicit_category;
567
568 if (sample_from_explicit_list && has_explicit_lights) { // --- Sample from explicit scene.lights list ---
569 int light_idx = static_cast<int>(local_rand01() * scene.lights.size());
570 light_idx = std::min(light_idx, (int)scene.lights.size() - 1);
571 const RayTracer::Light& L_obj = scene.lights[light_idx];
572
573 if (L_obj.type == RayTracer::LightType::Point) {
574 Le_value_out = Math::Vector3D(L_obj.intensity, L_obj.intensity, L_obj.intensity);
575 light_ray_out.origin = L_obj.position;
576 light_ray_out.direction = sample_uniform_sphere_direction_bdpt(pdf_emit_dir_solid_angle_out);
577
578 double p_sample_this_light_from_explicits = 1.0 / static_cast<double>(scene.lights.size());
579 pdf_pos_area_light_out = prob_pick_explicit_category * p_sample_this_light_from_explicits;
580
581 light_vertex_out.rec.point = L_obj.position;
582 light_vertex_out.rec.material.emission = Le_value_out;
583 light_vertex_out.rec.material.emissionStrength = 1.0;
584 light_vertex_out.rec.normal = Math::Vector3D(0,0,1);
585 light_vertex_out.is_point_light_source = true;
586 light_vertex_out.delta = true;
587 return true;
588 }
589 } else if (has_emissive_prims) { // --- Sample from emissive scene objects (area lights) ---
590 int emissive_obj_idx = static_cast<int>(local_rand01() * emissive_prims.size());
591 emissive_obj_idx = std::min(emissive_obj_idx, (int)emissive_prims.size() - 1);
592 const RayTracer::Primitive* emissive_prim = emissive_prims[emissive_obj_idx];
593
594 if (const RayTracer::Sphere* sph_light = dynamic_cast<const RayTracer::Sphere*>(emissive_prim)) {
595 double sphere_surface_area = 4.0 * M_PI * sph_light->radius * sph_light->radius;
596 if (sphere_surface_area < GEOMETRY_EPSILON) return false;
597
598 double p_sample_emissive_category = (has_explicit_lights && has_emissive_prims) ? (1.0 - prob_pick_explicit_category) : 1.0;
599 double p_this_prim_from_emissives = 1.0 / static_cast<double>(emissive_prims.size());
600 double p_point_on_surface_area = 1.0 / sphere_surface_area;
601 pdf_pos_area_light_out = p_sample_emissive_category * p_this_prim_from_emissives * p_point_on_surface_area;
602
603 double pdf_dummy_dir_sphere;
604 Math::Vector3D dir_from_center = sample_uniform_sphere_direction_bdpt(pdf_dummy_dir_sphere);
605 light_vertex_out.rec.point = sph_light->center + dir_from_center * sph_light->radius;
606 light_vertex_out.rec.normal = dir_from_center;
607 light_vertex_out.rec.material = sph_light->getMaterial();
608
609 Le_value_out = light_vertex_out.rec.material.emission * light_vertex_out.rec.material.emissionStrength;
610
611 light_ray_out.origin = light_vertex_out.rec.point + light_vertex_out.rec.normal * RAY_OFFSET_EPSILON;
612 light_ray_out.direction = sample_cosine_hemisphere_bdpt(light_vertex_out.rec.normal, pdf_emit_dir_solid_angle_out);
613
614 light_vertex_out.is_point_light_source = false;
615 light_vertex_out.delta = false;
616 return true;
617 }
618 // TODO: Add handling for other emissive primitive types (e.g., emissive quads, triangles from meshes).
619 }
620 return false;

633 double geometry_term_fixed(const RayTracer::Scene& scene, const RayTracer::Vertex& v0, const RayTracer::Vertex& v1) {
634 Math::Vector3D d_vec = v1.rec.point - v0.rec.point;
635 double dist_sq = d_vec.dot(d_vec);
636
637 if (dist_sq < GEOMETRY_EPSILON * GEOMETRY_EPSILON) return 0.0;
638
639 double dist = std::sqrt(dist_sq);
640 Math::Vector3D dir_v0_to_v1 = d_vec / dist; // Normalized direction from v0 to v1.
641
642 // --- Visibility Check (Shadow Ray) ---
643 Math::Point3D shadow_ray_origin_pt = v0.rec.point;
644 Math::Vector3D offset_normal_v0 = v0.rec.normal;
645 if (!v0.is_point_light_source) {
646 if (offset_normal_v0.dot(dir_v0_to_v1) < 0.0) offset_normal_v0 *= -1.0;
647 shadow_ray_origin_pt = shadow_ray_origin_pt + (offset_normal_v0 * RAY_OFFSET_EPSILON);
648 } else {
649 shadow_ray_origin_pt = shadow_ray_origin_pt + (dir_v0_to_v1 * RAY_OFFSET_EPSILON);
650 }
651
652 RayTracer::Ray shadow_ray(shadow_ray_origin_pt, dir_v0_to_v1);
653 RayTracer::HitRecord shadow_rec;
654 double t_max_shadow = dist - RAY_OFFSET_EPSILON * 2.0;
655
656 if (t_max_shadow > GEOMETRY_EPSILON) {
657 if (scene.hit(shadow_ray, t_max_shadow, shadow_rec)) {
658 return 0.0;
659 }
660 } else if (dist > RAY_OFFSET_EPSILON * 3.0) {
661 return 0.0;
662 }
663 // --- Cosine Factors ---
664 double cos0 = 1.0;
665 if (!v0.is_point_light_source && !v0.delta) {
666 Math::Vector3D N0_shading = v0.rec.normal;
667 cos0 = std::abs(N0_shading.dot(dir_v0_to_v1));
668 }
669
670 double cos1 = 1.0;
671 if (!v1.is_point_light_source && !v1.delta) {
672 if(v1.rec.t != std::numeric_limits<double>::infinity()){
673 Math::Vector3D N1_shading = v1.rec.normal;
674 cos1 = std::abs(N1_shading.dot(dir_v0_to_v1 * -1.0));
675 }
676 }
677 return (cos0 * cos1) / dist_sq;
678 }

688 inline double mis_balance(double p_cam, double p_light) {
689 double sum_pdfs = p_cam + p_light;
690 if (sum_pdfs < PDF_EPSILON) return 0.0;
691 return p_cam / sum_pdfs;
692 }

701 inline double mis_power(double p_cam, double p_light) {
702 double wc = p_cam * p_cam;
703 double wl = p_light * p_light;
704 double sum_weighted_pdfs = wc + wl;
705 if (sum_weighted_pdfs < PDF_EPSILON * PDF_EPSILON) return 0.0;
706 return wc / sum_weighted_pdfs;
707 }

728 double mis_weight_fixed(const Path& cam_path, size_t s_cam_idx, const Path& light_path, size_t t_light_idx, [[maybe_unused]] int k_total) {
729 if (s_cam_idx >= cam_path.size() && t_light_idx >= light_path.size()) return 1.0;
730
731 double p_cam_strategy = 1.0;
732 for (size_t i = 0; i <= s_cam_idx && i < cam_path.size(); ++i) {
733 p_cam_strategy *= (cam_path[i].pdfFwd + PDF_EPSILON);
734 }
735 for (size_t i = 0; i <= t_light_idx && i < light_path.size(); ++i) {
736 if (i > 0 && i < light_path.size()) {
737 p_cam_strategy *= (light_path[t_light_idx - i].pdfRev + PDF_EPSILON);
738 }
739 }
740
741
742 double p_light_strategy = 1.0;
743 for (size_t i = 0; i <= t_light_idx && i < light_path.size(); ++i) {
744 p_light_strategy *= (light_path[i].pdfFwd + PDF_EPSILON);
745 }
746 for (size_t i = 0; i <= s_cam_idx && i < cam_path.size(); ++i) {
747 if (i > 0 && i < cam_path.size()) {
748 p_light_strategy *= (cam_path[s_cam_idx - i].pdfRev + PDF_EPSILON);
749 }
750 }
751
752 // Power heuristic (beta=2).
753 double wc = p_cam_strategy * p_cam_strategy;
754 double wl = p_light_strategy * p_light_strategy;
755 double denom = wc + wl;
756
757 if (denom < (PDF_EPSILON * PDF_EPSILON) ) return 0.0;
758 return wc / denom;
759 }

776 std::vector<Math::Vector3D> renderBDPT(const RayTracer::Scene& scene,
777 const Utils::RenderParams& params,
778 bool showProgress) {
779 std::vector<Math::Vector3D> framebuffer(params.W * params.H, Math::Vector3D(0,0,0));
780 int max_total_bounces = params.max_depth;
781
782 double exposure_factor = 0.5; // Simple exposure control. Adjust for overall brightness.
783
784 #ifdef _OPENMP
785 #pragma omp parallel for schedule(dynamic, 1)
786 #endif
787 for (int j = 0; j < params.H; ++j) { // Iterate over image rows.
788 if (omp_get_thread_num() == 0 && showProgress && (j % 10 == 0 || j == params.H -1) ) { // Progress indicator from one thread.
789 std::cerr << "\rScanlines remaining (BDPT): " << (params.H - 1 - j) << " / " << params.H << " " << std::flush;
790 }
791 for (int i = 0; i < params.W; ++i) {
792 Math::Vector3D pixel_color_accumulator(0,0,0);
793 for (int sample_idx = 0; sample_idx < params.spp; ++sample_idx) { // Samples per pixel (SPP).
794 // Generate camera ray with jitter for anti-aliasing.
795 double u_jitter = (static_cast<double>(i) + local_rand01()) / (static_cast<double>(params.W));
796 double v_jitter = (static_cast<double>(params.H - 1 - j) + local_rand01()) / (static_cast<double>(params.H)); // Y is often flipped.
797 RayTracer::Ray cam_ray = scene.camera.ray(u_jitter, v_jitter);
798
799 // --- Generate Camera Sub-path (y_0, ..., y_s_max) ---
800 Path camera_path_vertices;
801 generate_subpath_fixed(scene, cam_ray, max_total_bounces, camera_path_vertices, true, params);
802
803 // --- Generate Light Sub-path (z_0, ..., z_t_max) ---
804 RayTracer::Ray initial_light_ray;
805 Math::Vector3D Le_raw_sampled_light_intensity;
806 double pdf_pos_area_light = 0.0;
807 double pdf_emit_dir_solid_angle = 0.0;
808 RayTracer::Vertex light_emitter_vertex;
809
810 bool light_source_was_sampled = sample_light_source_fixed(scene, initial_light_ray, Le_raw_sampled_light_intensity,
811 pdf_pos_area_light, pdf_emit_dir_solid_angle, light_emitter_vertex);
812
813 Path light_path_from_source_segments;
814 Path full_light_path_vertices;
815
816 if (light_source_was_sampled) {
817 double combined_initial_light_pdf_A_times_W = pdf_pos_area_light * pdf_emit_dir_solid_angle;
818 if (combined_initial_light_pdf_A_times_W < PDF_EPSILON) {
819 light_emitter_vertex.beta = Math::Vector3D(0,0,0);
820 } else {
821 light_emitter_vertex.beta = Le_raw_sampled_light_intensity / combined_initial_light_pdf_A_times_W;
822 }
823 light_emitter_vertex.pdfFwd = combined_initial_light_pdf_A_times_W;
824 light_emitter_vertex.pdfRev = 0;
825 full_light_path_vertices.push_back(light_emitter_vertex);
826 if (light_emitter_vertex.beta.dot(light_emitter_vertex.beta) > THROUGHPUT_EPSILON_SQ) {
827 generate_subpath_fixed(scene, initial_light_ray, max_total_bounces - 1,
828 light_path_from_source_segments, false, params);
829
830 for(const auto& segment_v_data : light_path_from_source_segments) {
831 RayTracer::Vertex actual_segment_v = segment_v_data;
832 actual_segment_v.beta = light_emitter_vertex.beta * segment_v_data.beta; // Accumulate throughput: beta(z_0...z_i) = beta(z_0) * beta(z_1...z_i).
833 full_light_path_vertices.push_back(actual_segment_v);
834 }
835 }
836 }
837 Math::Vector3D L_for_sample(0,0,0);
838 // === Accumulate Contributions from Different BDPT Strategies ===
839 for (size_t s_idx = 0; s_idx < camera_path_vertices.size(); ++s_idx) {
840 const auto& cam_v = camera_path_vertices[s_idx];
841 Math::Vector3D emission_val = cam_v.rec.material.emission * cam_v.rec.material.emissionStrength;
842 if (emission_val.dot(emission_val) > GEOMETRY_EPSILON) {
843 double w = mis_weight_fixed(camera_path_vertices, s_idx, Path{}, (size_t)-1, s_idx);
844 L_for_sample += cam_v.beta * emission_val * w;
845 }
846 }
847
848 if (!camera_path_vertices.empty() && !full_light_path_vertices.empty()) {
849 for (size_t s_idx = 0; s_idx < camera_path_vertices.size(); ++s_idx) {
850 const auto& cam_v = camera_path_vertices[s_idx];
851 if (cam_v.delta || cam_v.rec.t == std::numeric_limits<double>::infinity()) continue;
852
853 const auto& light_v_emitter = full_light_path_vertices[0];
854 if (light_v_emitter.beta.dot(light_v_emitter.beta) < THROUGHPUT_EPSILON_SQ) continue;
855 double G = geometry_term_fixed(scene, cam_v, light_v_emitter);
856 if (G < GEOMETRY_EPSILON) continue;
857
858 Math::Vector3D dir_ys_to_z0 = safe_normalize(light_v_emitter.rec.point - cam_v.rec.point);
859 Math::Vector3D bsdf_at_ys = eval_bsdf(cam_v.rec, cam_v.wi /* outgoing from y_s towards camera */,
860 dir_ys_to_z0 /* incoming to y_s from z_0 */,
861 cam_v.rec.normal, false);
862 Math::Vector3D L_connect_explicit = cam_v.beta * bsdf_at_ys * G * light_v_emitter.beta;
863
864 // MIS weight for strategy (s,1) using Veach's notation (s camera edges, 1 implicit light edge z_0 to eye).
865 double w_connect_explicit = mis_weight_fixed(camera_path_vertices, s_idx, full_light_path_vertices, 0, s_idx + 1);
866 L_for_sample += L_connect_explicit * w_connect_explicit;
867 }
868 for (size_t s_idx = 0; s_idx < camera_path_vertices.size(); ++s_idx) {
869 const auto& cam_v = camera_path_vertices[s_idx]; // Vertex y_s.
870 if (cam_v.delta || cam_v.rec.t == std::numeric_limits<double>::infinity()) continue; // Cannot connect from delta/sky.
871 for (size_t t_idx = 1; t_idx < full_light_path_vertices.size(); ++t_idx) {
872 const auto& light_v = full_light_path_vertices[t_idx];
873 if (light_v.delta || light_v.rec.t == std::numeric_limits<double>::infinity()) continue;
874 if (light_v.beta.dot(light_v.beta) < THROUGHPUT_EPSILON_SQ) continue;
875
876 double G = geometry_term_fixed(scene, cam_v, light_v);
877 if (G < GEOMETRY_EPSILON) continue;
878 Math::Vector3D dir_ys_to_zt = safe_normalize(light_v.rec.point - cam_v.rec.point);
879 Math::Vector3D bsdf_at_ys = eval_bsdf(cam_v.rec, cam_v.wi, dir_ys_to_zt, cam_v.rec.normal, false);
880 Math::Vector3D bsdf_at_zt = eval_bsdf(light_v.rec, light_v.wi, dir_ys_to_zt * -1.0, light_v.rec.normal, false);
881
882 Math::Vector3D L_connect = cam_v.beta * bsdf_at_ys * G * bsdf_at_zt * light_v.beta;
883 double w = mis_weight_fixed(camera_path_vertices, s_idx, full_light_path_vertices, t_idx, s_idx + t_idx + 1);
884 L_for_sample += L_connect * w;
885 }
886 }
887 }
888 pixel_color_accumulator += L_for_sample;
889 }
890
891 Math::Vector3D final_color_hdr = pixel_color_accumulator / static_cast<double>(params.spp);
892 final_color_hdr = final_color_hdr * exposure_factor; // Apply exposure.
893
894 // Reinhard tone mapping.
895 Math::Vector3D final_color_tm = final_color_hdr / (final_color_hdr + Math::Vector3D(1.0, 1.0, 1.0));
896
897 // Simple gamma correction (approximate sRGB).
898 final_color_tm.x = std::pow(final_color_tm.x, 1.0/2.2);
899 final_color_tm.y = std::pow(final_color_tm.y, 1.0/2.2);
900 final_color_tm.z = std::pow(final_color_tm.z, 1.0/2.2);
901
902 // Clamp color values to [0,1]
903 final_color_tm.x = std::max(0.0, std::min(1.0, final_color_tm.x));
904 final_color_tm.y = std::max(0.0, std::min(1.0, final_color_tm.y));
905 final_color_tm.z = std::max(0.0, std::min(1.0, final_color_tm.z));
906
907 framebuffer[j * params.W + i] = final_color_tm;
908 }
909 }
910 if (showProgress) std::cerr << "\nBDPT rendering finished." << std::endl;
911 return framebuffer;
912 }
