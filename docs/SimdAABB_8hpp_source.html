36 // Carnage-level 8-way AABB intersection with early exits and FMA
37 inline __m256i hitFast8(
38 const __m256& origX, const __m256& origY, const __m256& origZ,
39 const __m256& invX, const __m256& invY, const __m256& invZ,
40 const __m256i& negX, const __m256i& negY, const __m256i& negZ,
41 const AABB8& b,
42 __m256& tmin, __m256& tmax
43 ) {
44 // Precompute origin * invDir
45 __m256 oix = _mm256_mul_ps(origX, invX);
46 __m256 oiy = _mm256_mul_ps(origY, invY);
47 __m256 oiz = _mm256_mul_ps(origZ, invZ);
48
49 // X axis slab
50 __m256 bx0 = _mm256_blendv_ps(_mm256_load_ps(b.minX), _mm256_load_ps(b.maxX), _mm256_castsi256_ps(negX));
51 __m256 bx1 = _mm256_blendv_ps(_mm256_load_ps(b.maxX), _mm256_load_ps(b.minX), _mm256_castsi256_ps(negX));
52 __m256 tx0 = _mm256_fmsub_ps(bx0, invX, oix);
53 __m256 tx1 = _mm256_fmsub_ps(bx1, invX, oix);
54 tmin = _mm256_max_ps(tmin, tx0);
55 tmax = _mm256_min_ps(tmax, tx1);
56 // early out if nothing remains
57 __m256i m = _mm256_castps_si256(_mm256_cmp_ps(tmax, tmin, _CMP_GT_OQ));
58 if (_mm256_testz_si256(m, m)) return m;
59
60 // Y axis slab
61 __m256 by0 = _mm256_blendv_ps(_mm256_load_ps(b.minY), _mm256_load_ps(b.maxY), _mm256_castsi256_ps(negY));
62 __m256 by1 = _mm256_blendv_ps(_mm256_load_ps(b.maxY), _mm256_load_ps(b.minY), _mm256_castsi256_ps(negY));
63 __m256 ty0 = _mm256_fmsub_ps(by0, invY, oiy);
64 __m256 ty1 = _mm256_fmsub_ps(by1, invY, oiy);
65 tmin = _mm256_max_ps(tmin, ty0);
66 tmax = _mm256_min_ps(tmax, ty1);
67 m = _mm256_castps_si256(_mm256_cmp_ps(tmax, tmin, _CMP_GT_OQ));
68 if (_mm256_testz_si256(m, m)) return m;
69
70 // Z axis slab
71 __m256 bz0 = _mm256_blendv_ps(_mm256_load_ps(b.minZ), _mm256_load_ps(b.maxZ), _mm256_castsi256_ps(negZ));
72 __m256 bz1 = _mm256_blendv_ps(_mm256_load_ps(b.maxZ), _mm256_load_ps(b.minZ), _mm256_castsi256_ps(negZ));
73 __m256 tz0 = _mm256_fmsub_ps(bz0, invZ, oiz);
74 __m256 tz1 = _mm256_fmsub_ps(bz1, invZ, oiz);
75 tmin = _mm256_max_ps(tmin, tz0);
76 tmax = _mm256_min_ps(tmax, tz1);
77
78 // Final mask
79 __m256 mask = _mm256_cmp_ps(tmax, tmin, _CMP_GT_OQ);
80 return _mm256_castps_si256(mask);
81 }
